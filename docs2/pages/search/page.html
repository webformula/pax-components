  document.querySelector('mdw-search#two').addEventListener('search', event => {
    event.target.updateQuickResults([{
      value: 'one',
      primary: 'One',
    },{
      value: 'two',
      primary: 'Two'
    }].filter(v => v.primary.toLocaleLowerCase().includes(event.target.searchValue)));
  });

  // text input
  document.querySelector('mdw-search#two').addEventListener('input', () => {});

  // search result selected
  document.querySelector('mdw-search#two').addEventListener('change', event => {
    console.log(event.target.value);
  });
  `}
